  template< \
  typename T, \

public: \
  Property_##CLASS_NAME##_##NAME() {} \

  T &operator=(const T &i) {  \

  T value;  \

class PropertySample {
public:
  PropertySample() : c(0) {}

  int get_c() {
    int result = c.value * 3;
    return result;
  }

    return c * 3;

  std::string get_str() {
